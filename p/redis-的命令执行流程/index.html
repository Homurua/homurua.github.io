int _writeToClient(client *c, ssize_t *nwritten) {
    *nwritten = 0;
    if (c->bufpos > 0) {
        *nwritten = connWrite(c->conn, c->buf + c->sentlen, c->bufpos - c->sentlen);
        if (*nwritten <= 0)
            return C_ERR;
        c->sentlen += *nwritten;

        if ((int)c->sentlen == c->bufpos) {
            c->bufpos = 0;
            c->sentlen = 0;
        }
    }

    return C_OK;
}

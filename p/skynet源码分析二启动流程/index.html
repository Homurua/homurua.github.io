void skynet_start(struct skynet_config *config) {
    // register SIGHUP for log file reopen
    struct sigaction sa;
    sa.sa_handler = &handle_hup;
    sa.sa_flags = SA_RESTART;
    sigfillset(&sa.sa_mask);
    sigaction(SIGHUP, &sa, NULL);

    if (config->daemon) {
        if (daemon_init(config->daemon)) {
            exit(1);
        }
    }
    // åˆå§‹åŒ–èŠ‚ç‚¹æ•°é‡
    skynet_harbor_init(config->harbor);
    // åˆå§‹åŒ– handle å­˜å‚¨å™¨
    skynet_handle_init(config->harbor);
    // åˆå§‹åŒ–å…¨å±€æ¶ˆæ¯é˜Ÿåˆ—
    skynet_mq_init();
    // åˆå§‹åŒ– C æ¨¡å—ç®¡ç†å™¨ï¼Œè®¾ç½®æŸ¥æ‰¾è·¯å¾„
    skynet_module_init(config->module_path);
    // åˆå§‹åŒ–å…¨å±€æ—¶é—´
    skynet_timer_init();
    // åˆå§‹åŒ– socket ç®¡ç†å™¨
    skynet_socket_init();
    // æ ‡è®°æ˜¯å¦å¼€äº†æ€§èƒ½æµ‹è¯•
    skynet_profile_enable(config->profile);

    // åˆ›å»º logger CæœåŠ¡
    struct skynet_context *ctx = skynet_context_new(config->logservice, config->logger);
    if (ctx == NULL) {
        fprintf(stderr, "Can't launch %s service\n", config->logservice);
        exit(1);
    }
    // æ³¨å†Œ logger æœåŠ¡çš„åå­—
    skynet_handle_namehandle(skynet_context_handle(ctx), "logger");

    // å¯åŠ¨ï¼Œæ‰§è¡Œ config æ–‡ä»¶ä¸­çš„ bootstrap å‘½ä»¤
    bootstrap(ctx, config->bootstrap);

    // å¯åŠ¨å…¨éƒ¨çº¿ç¨‹
    start(config->thread);

    // harbor_exit may call socket send, so it should exit before socket_free
    skynet_harbor_exit();
    skynet_socket_free();
    if (config->daemon) {
        daemon_exit(config->daemon);
    }
}

static void bootstrap(struct skynet_context *logger, const char *cmdline) {
    int sz = strlen(cmdline);
    char name[sz + 1];
    char args[sz + 1];
    sscanf(cmdline, "%s %s", name, args); // ä» cmdline ä¸­è¯»å–æœåŠ¡åå­—å’Œå®ƒçš„å‚æ•°
    // åˆ›å»ºåˆå§‹æœåŠ¡

    if (ctx == NULL) {
        skynet_error(NULL, "Bootstrap error : %s\n", cmdline);
        skynet_context_dispatchall(logger);
        exit(1);
    }
}
